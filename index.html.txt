<script>
    // ЗВУКИ (синтезированные, без внешних файлов)
    const AudioContext = window.AudioContext || window.webkitAudioContext;
    let audioCtx = null;
    function initAudio() {
        if (!audioCtx) {
            audioCtx = new AudioContext();
        }
        if (audioCtx.state === 'suspended') {
            audioCtx.resume();
        }
    }
    function playBeep(freq, duration, type='sine', volume=0.2) {
        if (!audioCtx) return;
        try {
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.connect(gain);
            gain.connect(audioCtx.destination);
            osc.frequency.value = freq;
            osc.type = type;
            gain.gain.value = volume;
            osc.start();
            gain.gain.exponentialRampToValueAtTime(0.00001, audioCtx.currentTime + duration);
            osc.stop(audioCtx.currentTime + duration);
        } catch(e) {}
    }
    function playWin() { playBeep(880,0.3,'sine',0.3); setTimeout(()=>playBeep(1200,0.5,'sine',0.3),200); }
    function playLose() { playBeep(300,0.8,'sawtooth',0.4); setTimeout(()=>playBeep(200,0.6,'sawtooth',0.4),400); }
    function playFreddy() { playBeep(180,0.4,'square',0.35); setTimeout(()=>playBeep(150,0.8,'square',0.35),300); }
    function playComplaint() { playBeep(660,0.1,'sine',0.1); }
    function playClick() { playBeep(800,0.05,'sine',0.05); }
    function playTransition() { playBeep(540,0.2,'sine',0.2); }

    let currentNorma = 1, processed = 0, requiredForNorma = 0;
    let isFreddyComplaint = false, waitingForTemplate = false, timerInterval = null, timeRemaining = 0;
    let gameActive = true, gameWin = false;
    const normalComplaints = ["Помогите, застрял в текстуре!","Почините машину, она не едет!","Помогите, упал в воду!","Сломалась анимация персонажа","У меня украли внутриигровую валюту!","Не могу зайти в игру, вылетает","Звук пропал после обновления","Квест не завершается, баг"];
    function getFreddyChance() { return {1:5,2:12,3:30,4:55,5:75}[currentNorma] || 5; }
    function getFailChanceOnFreddy() { return {1:20,2:35,3:50,4:70,5:85}[currentNorma] || 50; }
    function updateRequiredForNorma() {
        if (currentNorma === 1 || currentNorma === 2) requiredForNorma = 5;
        else if (currentNorma === 3) requiredForNorma = 7;
        else requiredForNorma = 10;
        document.getElementById("requiredSpan").innerText = requiredForNorma;
    }
    function saveProgress() { localStorage.setItem("freddy_save_norma", currentNorma); }
    function loadSavedNorma() { let s = localStorage.getItem("freddy_save_norma"); return (s && !isNaN(parseInt(s))) ? parseInt(s) : 1; }
    function clearSave() { localStorage.removeItem("freddy_save_norma"); }

    function startNormaTimer() {
        if (timerInterval) clearInterval(timerInterval);
        let totalSeconds = requiredForNorma * 10;
        timeRemaining = totalSeconds;
        const timerElem = document.getElementById("timerDisplay");
        function update() {
            let m = Math.floor(timeRemaining / 60), s = timeRemaining % 60;
            timerElem.innerText = `${m.toString().padStart(2,'0')}:${s.toString().padStart(2,'0')}`;
        }
        update();
        timerInterval = setInterval(() => {
            if (!gameActive || waitingForTemplate || gameWin) return;
            if (timeRemaining <= 0) {
                if (processed < requiredForNorma && !gameWin && gameActive) loseGame("Время вышло! Вы не успели обработать норму. Фредди написал жбан на форум.");
                clearInterval(timerInterval); timerInterval = null;
            } else { timeRemaining--; update(); }
        }, 1000);
    }

    function generateNewComplaint() {
        if (!gameActive || gameWin) return;
        if (processed >= requiredForNorma) { completeNorma(); return; }
        let isFreddy = (Math.random() * 100) < getFreddyChance();
        if (isFreddy) {
            isFreddyComplaint = true;
            document.getElementById("complaintText").innerHTML = "🐻❄️ ФРЕДДИ: Твоя работа меня бесит! ОТВЕЧАЙ, НО ОСТОРОЖНЕЕ... 🐻❄️";
            playFreddy();
        } else {
            isFreddyComplaint = false;
            let randomIndex = Math.floor(Math.random() * normalComplaints.length);
            document.getElementById("complaintText").innerHTML = "📢 ЖАЛОБА: " + normalComplaints[randomIndex];
            playComplaint();
        }
        document.getElementById("templatesPanel").style.display = "none";
        waitingForTemplate = false;
        let mainBtn = document.getElementById("mainAnswerBtn");
        mainBtn.style.display = "block";
        mainBtn.innerText = "📝 ОТВЕТИТЬ НА ЖАЛОБУ";
    }

    function onMainAnswer() { if(!gameActive||gameWin||waitingForTemplate) return; playClick(); waitingForTemplate=true; document.getElementById("templatesPanel").style.display="flex"; document.getElementById("mainAnswerBtn").style.display="none"; }
    function onSelectTemplate(replyType) {
        if(!gameActive||gameWin||!waitingForTemplate) return;
        waitingForTemplate=false; document.getElementById("templatesPanel").style.display="none";
        if(isFreddyComplaint) {
            let fail=(Math.random()*100)<getFailChanceOnFreddy();
            if(fail){ loseGame("😱 Фредди в ярости! Ваш ответ ему не понравился. Он написал жалобу на форум. ИГРА НАЧАЛАСЬ ЗАНОВО."); return; }
            else { document.getElementById("complaintText").innerHTML="😅 Фредди нехотя принял ответ. Жалоба закрыта."; processSuccessfulComplaint(); return; }
        } else { processSuccessfulComplaint(); }
    }
    function processSuccessfulComplaint() {
        if(!gameActive||gameWin) return;
        processed++; document.getElementById("processedCount").innerText=processed;
        if(processed>=requiredForNorma) completeNorma();
        else setTimeout(()=>{ if(gameActive&&!gameWin) generateNewComplaint(); },800);
    }
    function completeNorma() {
        if(currentNorma===5){
            gameWin=true; gameActive=false; if(timerInterval)clearInterval(timerInterval);
            document.getElementById("complaintText").innerHTML="🏆 ПОБЕДА! 🏆<br>Вы прошли 5 норм!";
            document.getElementById("mainAnswerBtn").style.display="none"; document.getElementById("templatesPanel").style.display="none";
            document.getElementById("messageBox").innerHTML="✨ Вы справились! Фредди бессилен ✨";
            clearSave(); playWin(); return;
        }
        currentNorma++; processed=0; updateRequiredForNorma(); saveProgress();
        document.getElementById("normaCount").innerText=currentNorma; document.getElementById("processedCount").innerText="0";
        document.getElementById("complaintText").innerHTML=`⭐ ПЕРЕХОД НА ${currentNorma} НОРМУ ⭐<br>Теперь нужно ответить на ${requiredForNorma} жалоб!`;
        if(timerInterval)clearInterval(timerInterval); startNormaTimer(); playTransition();
        setTimeout(()=>{ if(gameActive&&!gameWin) generateNewComplaint(); },2000);
    }
    function loseGame(reason) {
        if(!gameActive) return;
        gameActive=false; if(timerInterval)clearInterval(timerInterval);
        document.getElementById("complaintText").innerHTML=`💀 ЖБАН НА ФОРУМЕ 💀<br>${reason}`;
        document.getElementById("mainAnswerBtn").style.display="none"; document.getElementById("templatesPanel").style.display="none";
        document.getElementById("messageBox").innerHTML="❌ Игра окончена. Нажми «В главное меню» ❌";
        clearSave(); playLose();
    }
    function startGameFromNorma(normaValue) {
        if(timerInterval)clearInterval(timerInterval);
        gameActive=true; gameWin=false; currentNorma=normaValue; processed=0; updateRequiredForNorma();
        isFreddyComplaint=false; waitingForTemplate=false;
        document.getElementById("normaCount").innerText=currentNorma; document.getElementById("processedCount").innerText="0";
        document.getElementById("messageBox").innerHTML=""; document.getElementById("complaintText").innerHTML=`Начинаем ${currentNorma} норму! Нужно ${requiredForNorma} жалоб.`;
        document.getElementById("mainAnswerBtn").style.display="block"; document.getElementById("templatesPanel").style.display="none";
        startNormaTimer();
        setTimeout(()=>{ if(gameActive) generateNewComplaint(); },1500);
        saveProgress();
    }
    function newGame() { clearSave(); startGameFromNorma(1); showGameArea(); }
    function continueGame() { let saved=loadSavedNorma(); if(saved>=1&&saved<=5) startGameFromNorma(saved); else newGame(); showGameArea(); }
    function showGameArea() { document.getElementById("mainMenu").style.display="none"; document.getElementById("gameArea").style.display="block"; initAudio(); }
    function showMainMenu() { if(timerInterval)clearInterval(timerInterval); document.getElementById("mainMenu").style.display="flex"; document.getElementById("gameArea").style.display="none"; gameActive=false; gameWin=false; }

    document.getElementById("newGameBtn").addEventListener("click", newGame);
    document.getElementById("continueBtn").addEventListener("click", continueGame);
    document.getElementById("exitToMenuBtn").addEventListener("click", showMainMenu);
    document.getElementById("exitBtn").addEventListener("click", ()=>{ alert("Закройте вкладку браузера, чтобы выйти"); });
    let aboutVisible=false;
    document.getElementById("aboutBtn").addEventListener("click",()=>{ let a=document.getElementById("aboutText"); aboutVisible=!aboutVisible; a.style.display=aboutVisible?"block":"none"; });
    document.getElementById("mainAnswerBtn").addEventListener("click", onMainAnswer);
    document.querySelectorAll(".template-btn").forEach(btn=>{ btn.addEventListener("click",(e)=>{ onSelectTemplate(btn.getAttribute("data-reply")); }); });
    showMainMenu();
</script>
